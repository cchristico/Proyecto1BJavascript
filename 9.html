Modules export and import

ES6 provee la funcionalidad en Javascript de construir modulos, los cuales contienen un porcion de codigo a ser ejecutada. Estos pueden ser importados o exportados de acuerdo a lo que se necesite en el codigo. 

    //------ lib.js ------
    export const sqrt = Math.sqrt;
    export function square(x) {
        return x * x;
    }
    export function diag(x, y) {
        return sqrt(square(x) + square(y));
    }
    
    //------ main.js ------
    import { square, diag } from 'lib';
    console.log(square(11)); // 121
    console.log(diag(4, 3)); // 5

import
       //------ main.js ------
    import * as lib from 'lib';
    console.log(lib.square(11)); // 121
    console.log(lib.diag(4, 3)); // 5
